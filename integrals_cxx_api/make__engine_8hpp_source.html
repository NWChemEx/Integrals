33 template<typename OpType>
34 auto make_engine(const std::vector<libint2::BasisSet>& bases, const OpType& op,
35  double thresh, std::size_t deriv = 0) {
36  // Variables for engine construction
37  constexpr auto libint_op = integrals::libint::detail_::op_v<OpType>;
38  auto max_nprims = libint2::max_nprim(bases[0]);
39  auto max_l = libint2::max_l(bases[0]);
40 
41  // Find max_nprims and max_l in bases
42  for(auto set : bases) {
43  max_nprims = std::max(max_nprims, libint2::max_nprim(set));
44  max_l = std::max(max_l, libint2::max_l(set));
45  }
46 
47  // Construct engine and handle specialized settings
48  if(!libint2::initialized()) libint2::initialize();
49  libint2::Engine engine(libint_op, max_nprims, max_l, deriv, thresh);
50  // Libint not acknowleding max_nprim in ctor?
51  engine.set_max_nprim(max_nprims);
52 
53  if(libint2::rank(libint_op) == 2) {
54  if(bases.size() == 2) engine.set(libint2::BraKet::xs_xs);
55  if(bases.size() == 3) engine.set(libint2::BraKet::xs_xx);
56  }
57 
58  if constexpr(std::is_same_v<OpType, simde::type::v_en_type>) {
59  const auto& nuclei = op.rhs_particle();
60  std::vector<std::pair<double, std::array<double, 3>>> qs;
61  for(const auto& ai : nuclei) {
62  std::array<double, 3> coords{ai.x(), ai.y(), ai.z()};
63  qs.emplace_back(static_cast<const double&>(ai.Z()), coords);
64  }
65  engine.set_params(qs);
66  }
67 
68  return engine;
69 }
