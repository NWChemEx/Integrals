29 inline auto shells2ord(const std::vector<libint2::BasisSet>& bases,
30  std::vector<std::size_t>& shells) {
31  using size_vector_t = std::vector<std::size_t>;
32 
33  // Calculate the ordinal step of each basis dimension other than the Nth.
34  auto N = bases.size();
35  size_vector_t step_sizes(N - 1);
36  for(decltype(N) i = 1; i <= N - 1; ++i) {
37  step_sizes[N - 1 - i] = bases[N - i].nbf();
38  if(i > 1) step_sizes[N - 1 - i] *= step_sizes[N - i];
39  }
40 
41  // Set starting and ending AO indices
42  size_vector_t curr_ao(N);
43  size_vector_t up_ao(N);
44  for(decltype(N) i = 0; i < N; ++i) {
45  curr_ao[i] = bases[i].shell2bf()[shells[i]];
46  up_ao[i] = curr_ao[i] + bases[i][shells[i]].size();
47  }
48 
49  // Increment through the AO indices of the shell and determine the ordinal
50  // index for each.
51  size_vector_t ords;
52  while(curr_ao[0] < up_ao[0]) {
53  // ordinal calculation
54  auto curr_ord = curr_ao.back();
55  for(std::size_t i = 0; i < step_sizes.size(); ++i) {
56  curr_ord += curr_ao[i] * step_sizes[i];
57  }
58  ords.push_back(curr_ord);
59 
60  // Increment to the next AO index.
61  curr_ao.back() += 1;
62  for(decltype(N) i = 1; i < N; ++i) {
63  if(curr_ao[N - i] >= up_ao[N - i]) {
64  // curr_ao[0] accumalates until it passes up_aos[0]
65  // and the loop terminates.
66  curr_ao[N - i] = bases[N - i].shell2bf()[shells[N - i]];
67  curr_ao[N - i - 1] += 1;
68  }
69  }
70  }
71  return ords;
72 }
