32template<typename T>
33auto get_permutations(T&& abcd, bool p0_1, bool p2_3, bool p01_23) {
34 using element_type = std::decay_t<decltype(abcd[0])>;
35 using index_type = std::array<element_type, 4>;
36
37 std::set<index_type> permutations{abcd};
38
39 if(p0_1) permutations.insert({abcd[1], abcd[0], abcd[2], abcd[3]});
40 if(p2_3) permutations.insert({abcd[0], abcd[1], abcd[3], abcd[2]});
41 if(p01_23) permutations.insert({abcd[2], abcd[3], abcd[0], abcd[1]});
42 if(p0_1 && p01_23)
43 permutations.insert({abcd[2], abcd[3], abcd[1], abcd[0]});
44 if(p2_3 && p01_23)
45 permutations.insert({abcd[3], abcd[2], abcd[0], abcd[1]});
46 if(p0_1 && p2_3) permutations.insert({abcd[1], abcd[0], abcd[3], abcd[2]});
47 if(p0_1 && p2_3 && p01_23)
48 permutations.insert({abcd[3], abcd[2], abcd[1], abcd[0]});
49
50 return permutations;
51}

62template<typename T>
63auto get_permutations_with_sigma(T&& abcd, bool p0_1, bool p2_3, bool p01_23) {
64 using element_type = std::decay_t<decltype(abcd[0])>;
65 using index_type = std::array<element_type, 4>;
66 using sigma_type = std::array<std::size_t, 4>;
67 using pair_type = std::pair<index_type, sigma_type>;
68
69 std::map<index_type, sigma_type> seen;
70 auto try_insert = [&](index_type perm, sigma_type sigma) {
71 seen.emplace(perm, sigma);
72 };
73
74 try_insert({abcd[0], abcd[1], abcd[2], abcd[3]}, {0, 1, 2, 3});
75 if(p0_1) try_insert({abcd[1], abcd[0], abcd[2], abcd[3]}, {1, 0, 2, 3});
76 if(p2_3) try_insert({abcd[0], abcd[1], abcd[3], abcd[2]}, {0, 1, 3, 2});
77 if(p01_23) try_insert({abcd[2], abcd[3], abcd[0], abcd[1]}, {2, 3, 0, 1});
78 if(p0_1 && p01_23)
79 try_insert({abcd[2], abcd[3], abcd[1], abcd[0]}, {2, 3, 1, 0});
80 if(p2_3 && p01_23)
81 try_insert({abcd[3], abcd[2], abcd[0], abcd[1]}, {3, 2, 0, 1});
82 if(p0_1 && p2_3)
83 try_insert({abcd[1], abcd[0], abcd[3], abcd[2]}, {1, 0, 3, 2});
84 if(p0_1 && p2_3 && p01_23)
85 try_insert({abcd[3], abcd[2], abcd[1], abcd[0]}, {3, 2, 1, 0});
86
87 std::vector<pair_type> result;
88 result.reserve(seen.size());
89 for(auto& [perm, sigma] : seen) result.emplace_back(perm, sigma);
90 return result;
91}
