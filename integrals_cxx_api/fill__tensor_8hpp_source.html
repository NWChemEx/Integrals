48template<typename FloatType>
49auto build_eigen_buffer(const std::vector<libint2::BasisSet>& basis_sets,
50 parallelzone::runtime::RuntimeView& rv, double thresh) {
51 FloatType initial_value;
52 if constexpr(std::is_same_v<FloatType, double>) {
53 initial_value = 0.0;
54 } else { // Presumably sigma::UDouble
55 initial_value = FloatType(0.0, thresh);
56 }
57 auto N = basis_sets.size();
58 std::vector<decltype(N)> dims(N);
59 for(decltype(N) i = 0; i < N; ++i) dims[i] = basis_sets[i].nbf();
60
61 using namespace tensorwrapper;
62 using shape_t = shape::Smooth;
63
64 shape_t s{dims.begin(), dims.end()};
65 auto buffer = buffer::make_contiguous<FloatType>(s, initial_value);
66 return std::make_unique<buffer::Contiguous>(std::move(buffer));
67}

69template<std::size_t N, typename FloatType>
70auto fill_tensor(const std::vector<libint2::BasisSet>& basis_sets,
71 const chemist::qm_operator::OperatorBase& op,
72 parallelzone::runtime::RuntimeView& rv, double thresh) {
73 using size_type = decltype(N);
74
75 // Dimensional information
76 std::vector<size_type> dim_stepsizes(N, 1);
77 size_type num_shell_combinations = 1;
78
79 for(size_type i = 0; i < N; ++i) {
80 num_shell_combinations *= basis_sets[i].size();
81 for(size_type j = i; j < N - 1; ++j) {
82 dim_stepsizes[i] *= basis_sets[j].size();
83 }
84 }
85
86 // Make an engine for each thread
87 int num_threads = get_num_threads();
88 std::vector<libint2::Engine> engines(num_threads);
89 LibintVisitor visitor(basis_sets, thresh);
90 op.visit(visitor);
91 for(int i = 0; i != num_threads; ++i) { engines[i] = visitor.engine(); }
92
93 // Fill in values
94 auto pbuffer = build_eigen_buffer<FloatType>(basis_sets, rv, thresh);
95 auto data = pbuffer->get_mutable_data();
96 auto span = wtf::buffer::contiguous_buffer_cast<FloatType>(data);
97#ifdef _OPENMP
98#pragma omp parallel for
99#endif
100 for(size_type i_pair = 0; i_pair != num_shell_combinations; ++i_pair) {
101 auto thread_id = get_thread_num();
102
103 std::vector<size_type> shells(N);
104 auto shell_ord = i_pair;
105 for(size_type i = 0; i < N; ++i) {
106 shells[i] = shell_ord / dim_stepsizes[i];
107 shell_ord = shell_ord % dim_stepsizes[i];
108 }
109
110 detail_::run_engine_(engines[thread_id], basis_sets, shells,
111 std::make_index_sequence<N>());
112
113 const auto& buf = engines[thread_id].results();
114 auto vals = buf[0];
115 if(vals) {
116 auto ord = detail_::shells2ord(basis_sets, shells);
117 auto n_ord = ord.size();
118 for(decltype(n_ord) i_ord = 0; i_ord < n_ord; ++i_ord) {
119 auto update = span[ord[i_ord]] + vals[i_ord];
120 span[ord[i_ord]] = update;
121 }
122 }
123 }
124
125 auto pshape = pbuffer->layout().shape().clone();
126 return simde::type::tensor(std::move(pshape), std::move(pbuffer));
127}
128} // namespace

130template<std::size_t N>
131auto fill_tensor(const std::vector<libint2::BasisSet>& basis_sets,
132 const chemist::qm_operator::OperatorBase& op,
133 parallelzone::runtime::RuntimeView& rv, double thresh,
134 bool with_uq) {
135 simde::type::tensor t;
136 if(with_uq) {
137 if constexpr(integrals::type::has_sigma()) {
138 t = fill_tensor<N, type::uncertain_double>(basis_sets, op, rv,
139 thresh);
140 } else {
141 throw std::runtime_error("Sigma support not enabled!");
142 }
143 } else {
144 t = fill_tensor<N, double>(basis_sets, op, rv, thresh);
145 }
146 return t;
147}
