31inline auto make_libint_basis_set(const simde::type::ao_basis_set& bs) {
33 using atom_t = libint2::Atom;
34 using shell_t = libint2::Shell;
35 using basis_t = libint2::BasisSet;
36 using cont_t = libint2::Shell::Contraction;
37 using svec_d_t = libint2::svector<double>;
38 using conts_t = libint2::svector<cont_t>;
39 using centers_t = std::vector<atom_t>;
40 using atom_bases_t = std::vector<shell_t>;
41 using element_bases_t = std::vector<atom_bases_t>;
42
44 centers_t centers{};
45 element_bases_t element_bases{};
46
48 auto atom_ctor = [](int Z, double x, double y, double z) {
49 atom_t atom{};
50 atom.atomic_number = Z;
51 atom.x = x;
52 atom.y = y;
53 atom.z = z;
54 return atom;
55 };
56
58 std::array<double, 3> origin = {0.0, 0.0, 0.0};
59
61 auto n_bs = bs.size();
62 for(decltype(n_bs) abs_i = 0; abs_i < n_bs; ++abs_i) {
64 const auto& abs = bs[abs_i];
65 centers.push_back(atom_ctor(abs_i, abs.center().x(), abs.center().y(),
66 abs.center().z()));
67
69 atom_bases_t atom_bases{};
70 for(const auto&& shelli : abs) {
71 const auto nprims = shelli.n_primitives();
72 const auto prim0 = shelli.primitive(0);
73 const auto primN = shelli.primitive(nprims - 1);
74 const bool pure = shelli.pure() == chemist::ShellType::pure;
75 const int l = shelli.l();
76
77 svec_d_t alphas(&prim0.exponent(), &primN.exponent() + 1);
78 svec_d_t coefs(&prim0.coefficient(), &primN.coefficient() + 1);
79 conts_t conts{cont_t{l, pure, coefs}};
81 atom_bases.push_back(shell_t(alphas, conts, origin));
82 }
83 element_bases.push_back(atom_bases);
84 }
85
87 return basis_t(centers, element_bases);
88}
